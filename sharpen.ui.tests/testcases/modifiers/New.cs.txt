namespace modifiers
{
	internal class Base
	{
		public virtual void foo()
		{
		}

		public virtual void bar()
		{
		}

		public int fooBar;
	}

	internal abstract class Derived : modifiers.Base
	{
		new public int foo;

		new public void bar()
		{
			
			
			fooBar = 42;
		}

		new public abstract void fooBar();
	}
}
