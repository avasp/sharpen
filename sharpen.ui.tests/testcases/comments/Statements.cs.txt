namespace comments
{
	public class Statements
	{
		public static void foo()
		{
			
			int a = 42;
			
			try
			{
				
				
				bar(a);
			}
			catch (System.Exception)
			{
			}
			
			for (int i = 0; i < 10; ++i)
			{
				
				if (0 == i % 2)
				{
					
					continue;
				}
				
				break;
			}
			
			while (a > 0)
			{
				
				--a;
			}
		}

		public static int bar(int value)
		{
			if (value == 42)
			{
				
				throw new System.Exception();
			}
			
			return value;
		}
	}
